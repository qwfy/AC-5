56   # An encoder is just an RNN. 
57   # We feed it the input sequence, harvest its final state, 
58   # and feed that to the decoder. 
59   def build_encoder(source_vocab_size, source_embedding_dim, 
60                     rnn_size, rnn_layers, 
61                     batch_size, handles): 
62    
63     # Input to the encoder. 
64     # Each entry is a vocabulary id, 
65     # and every sequence (i.e. row) is padded to have the same length. 
66     # (batch_size, sequence_length) 
67     input_ids = tf.placeholder(tf.int32, (None, None), name=handles.input_ids) 
68    
69     # Length of each sequence, without counting the padding. 
70     # (batch_size,) 
71     sequence_lengths = tf.placeholder(tf.int32, (None,), name=handles.input_lengths) 
72    
73     # Embedded version of input_ids. 
74     # (batch_size, max_time, source_embedding_dim) 
75     # where max_time == tf.shape(input_ids)[1] == tf.reduce_max(sequence_lengths) 
76     inputs_embedded = tf.contrib.layers.embed_sequence( 
77       ids=input_ids, 
78       vocab_size=source_vocab_size, 
79       embed_dim=source_embedding_dim) 
80    
81     def build_cell(): 
82       # The cell advance one time step in one layer. 
83       cell = tf.nn.rnn_cell.LSTMCell( 
84         num_units=rnn_size, 
85         initializer=tf.random_uniform_initializer(-0.1, 0.1)) 
86       return cell 
87    
88     # Conceptially: 
89     # 
90     # multi_rnn_cell(input,  [layer1_state,     layer2_state,     ...]) 
91     # ->            (output, [new_layer1_state, new_layer2_state, ...]) 
92     # 
93     # i.e. advance one time step in each layer. 
94     multi_rnn_cell = tf.nn.rnn_cell.MultiRNNCell( 
95       [build_cell() for _ in range(rnn_layers)]) 
96    
97     zero_state = multi_rnn_cell.zero_state(batch_size, tf.float32) 
98    
99     # Advance sequence_lengths[i] time steps for layer i, for all i. 
100    # It is a design choice that we only interested in the final state. 
101    # (rnn_layers, 2=lstm_state_tuple_size, batch_size, rnn_size) 
102    _outputs, final_state = tf.nn.dynamic_rnn( 
103      cell=multi_rnn_cell, 
104      inputs=inputs_embedded, 
105      sequence_length=sequence_lengths, 
106      initial_state=zero_state) 
107   
108    return final_state

111  # %% 
112  # The decoder is another RNN, it is a design choice that 
113  # it has the same number of layers and size with the encoder RNN. 
114  # 
115  # The final state of the encoder is passed as the initial state of the decoder, 
116  # and a start of sequence (SOS) symbol, (its embedding vector, to be precise), 
117  # is used as the input at time step 0. 
118  # 
119  # The output at each time step is projected (using a dense layer) to have vocab_size logits. 
120  # When making inference, we sample from the output at time t, 
121  # and use it as the input at time t+1. 
122  # 
123  # When training, we discard the RNN output, 
124  # and feed it the truth corresponding to each time step. 
125  def build_decoder(encoder_state, 
126                    target_vocab_size, target_embedding_dim, 
127                    rnn_size, rnn_layers, 
128                    batch_size, max_decode_iterations, vocab_to_id, handles): 
129   
130    # Input sequence to the decoder, this is the truth, and only used during training. 
131    # The first element of each sequence is always the SOS symbol. 
132    # (batch_size, sequence_length) 
133    input_ids = tf.placeholder(tf.int32, (None, None), name=handles.decoder_input_ids) 
134   
135    # Length of input_ids, without counting padding. 
136    # (batch_size,) 
137    sequence_lengths = tf.placeholder(tf.int32, (None,), name=handles.decoder_input_lengths) 
138   
139    input_embeddings = tf.Variable(tf.random_uniform((target_vocab_size, target_embedding_dim))) 
140   
141    inputs_embedded = tf.nn.embedding_lookup( 
142      params=input_embeddings, 
143      ids=input_ids) 
144   
145    def build_cell(): 
146      cell = tf.nn.rnn_cell.LSTMCell( 
147        num_units=rnn_size, 
148        initializer=tf.random_uniform_initializer(-0.1, 0.1)) 
149      return cell 
150   
151    multi_rnn_cell = tf.nn.rnn_cell.MultiRNNCell( 
152      [build_cell() for _ in range(rnn_layers)]) 
153   
154    # Transform the RNN output to logits, so that we can sample from it. 
155    projection_layer = tf.layers.Dense( 
156      units=target_vocab_size, 
157      activation=None, 
158      use_bias=False) 
159   
160    def make_logits(helper, reuse): 
161      with tf.variable_scope('decoder', reuse=reuse): 
162        decoder = tf.contrib.seq2seq.BasicDecoder( 
163          cell=multi_rnn_cell, 
164          helper=helper, 
165          initial_state=encoder_state, 
166          output_layer=projection_layer) 
167   
168        final_outputs, _final_state, _final_sequence_length = tf.contrib.seq2seq.dynamic_decode( 
169          decoder=decoder, 
170          maximum_iterations=max_decode_iterations, 
171          impute_finished=True) 
172   
173        # (batch_size, max_time, vocab_size) 
174        return final_outputs.rnn_output 
175   
176    # The Helper is used to sample from logits, 
177    # you can swap one Helper with another to get different sampling behaviour. 
178    # 
179    # At time t, a TrainingHelper just read data from inputs[:, t], 
180    # and use it as the input at t . 
181    train_helper = tf.contrib.seq2seq.TrainingHelper( 
182      inputs=inputs_embedded, 
183      sequence_length=sequence_lengths) 
184   
185    # Greedy, as in GreedyEmbeddingHelper, usually means that a max() is taken. 
186    # So at time t, output at t-1, which has the shape (vocab_size,), 
187    # (considering only one sample), is sampled from, and is used as the input at time t, 
188    # i.e. input at t = lookup_embedding(embedding, argmax(output at t-1)) 
189    infer_helper = tf.contrib.seq2seq.GreedyEmbeddingHelper( 
190      embedding=input_embeddings, 
191      start_tokens=tf.tile(tf.constant([vocab_to_id['<SOS>']]), [batch_size]), 
192      end_token=vocab_to_id['<EOS>']) 
193   
194    train_logits = make_logits(train_helper, reuse=False) 
195   
196    # Use the same variables used in training. 
197    infer_logits = make_logits(infer_helper, reuse=True) 
198    tf.identity(infer_logits, name=handles.infer_logits) 
199   
200    return train_logits

224  def pad_right(lol, vocab_to_id): 
225    # Pad right with <PAD>, such that all rows have the same length. 
226    row_lengths = [len(row) for row in lol] 
227    max_length = np.max(row_lengths) 
228    arr = np.ndarray((len(lol), max_length)) 
229    arr.fill(vocab_to_id['<PAD>']) 
230    for i, length, row in zip(range(len(lol)), row_lengths, lol): 
231      arr[i, :length] = row 
232    return arr, row_lengths 
233   
234   
235  def push_sos(arr, vocab_to_id): 
236    # Prepend each row with <SOS>, and drop the last column, 
237    # such that the length is unchanged. 
238    soses = np.ndarray((arr.shape[0], 1)) 
239    soses.fill(vocab_to_id['<SOS>']) 
240    with_sos = np.concatenate([soses, arr], axis=1) 
241    no_last = with_sos[:, :-1] 
242    return no_last

244  def make_batch(batch_size, lower, upper): 
245    # Make one batch. 
246    xys = np.random.randint(low=lower, high=upper, size=(batch_size, 2)) 
247    zs = np.sum(xys, axis=1) 
248   
249    source_ids = [[vocab_to_id[char] for char in f'{xy[0]}+{xy[1]}'] 
250                  for xy in list(xys)] 
251    target_ids = [[vocab_to_id[char] for char in f'{z}'] + [vocab_to_id['<EOS>']] 
252                  for z in list(zs)] 
253   
254    padded_source_ids, source_lengths = pad_right(source_ids, vocab_to_id) 
255    padded_target_ids, target_lengths = pad_right(target_ids, vocab_to_id) 
256    decoder_input_ids = push_sos(padded_target_ids, vocab_to_id) 
257    batch = (padded_source_ids, source_lengths, 
258             padded_target_ids, target_lengths, 
259             decoder_input_ids) 
260    return batch

264  def get_feed(all_batches, graph, handles): 
265    for (padded_source_ids, source_lengths, 
266         padded_target_ids, target_lengths, 
267         decoder_input_ids 
268         ) in all_batches: 
269      yield {graph.get_tensor_by_name(f'{handles.batch_size}:0'): len(source_lengths), 
270             graph.get_tensor_by_name(f'{handles.input_ids}:0'): padded_source_ids, 
271             graph.get_tensor_by_name(f'{handles.input_lengths}:0'): source_lengths, 
272             graph.get_tensor_by_name(f'{handles.decoder_input_ids}:0'): decoder_input_ids, 
273             graph.get_tensor_by_name(f'{handles.decoder_input_lengths}:0'): target_lengths, 
274             graph.get_tensor_by_name(f'{handles.target_ids}:0'): padded_target_ids, 
275             graph.get_tensor_by_name(f'{handles.target_lengths}:0'): target_lengths, 
276             graph.get_tensor_by_name(f'{handles.max_decode_iterations}:0'): 2 * np.max(source_lengths)}

326  # Create the graph. 
327  def make_graph(handles, hparams): 
328    graph = tf.Graph() 
329    with graph.as_default(): 
330      batch_size = tf.placeholder(tf.int32, (), name=handles.batch_size) 
331   
332      encoder_final_state = build_encoder( 
333        source_vocab_size=hparams.source_vocab_size, 
334        source_embedding_dim=hparams.source_embedding_dim, 
335        rnn_size=hparams.rnn_size, 
336        rnn_layers=hparams.rnn_layers, 
337        batch_size=batch_size, 
338        handles=handles) 
339   
340      max_decode_iterations = tf.placeholder(tf.int32, (), name=handles.max_decode_iterations) 
341   
342      train_logits = build_decoder( 
343        encoder_state=encoder_final_state, 
344        target_vocab_size=hparams.target_vocab_size, 
345        target_embedding_dim=hparams.target_embedding_dim, 
346        rnn_size=hparams.rnn_size, 
347        rnn_layers=hparams.rnn_layers, 
348        batch_size=batch_size, 
349        max_decode_iterations=max_decode_iterations, 
350        vocab_to_id=vocab_to_id, 
351        handles=handles) 
352   
353      # Labels. So it has EOS tokens in it. 
354      # Used only during training. 
355      # (batch_size, target_sequence_length) 
356      target_ids = tf.placeholder(tf.int32, (None, None), name=handles.target_ids) 
357   
358      # Length of target_ids, without counting padding. 
359      # (batch_size,) 
360      target_lengths = tf.placeholder(tf.int32, (None,), name=handles.target_lengths) 
361   
362      # Since out target_ids is effectively of variant length, 
363      # we mask out those padding positions. 
364      loss_mask = tf.sequence_mask(lengths=target_lengths, dtype=tf.float32) 
365      train_loss_ = tf.contrib.seq2seq.sequence_loss( 
366        logits=train_logits, 
367        targets=target_ids, 
368        weights=loss_mask) 
369      train_loss = tf.identity(train_loss_, name=handles.train_loss) 
370   
371      optimizer = tf.train.AdamOptimizer(hparams.lr) 
372      gradients = optimizer.compute_gradients(train_loss) 
373      clipped_gradients = [(tf.clip_by_value(grad, -5., 5.), var) 
374                           for grad, var in gradients if grad is not None] 
375      optimizer.apply_gradients(clipped_gradients, name=handles.optimize) 
376   
377      summary_writer = tf.summary.FileWriter( 
378        logdir=hparams.tensorboard_dir, 
379        graph=tf.get_default_graph()) 
380      summary_writer.close() 
381   
382    return graph

397  def train(sess: tf.Session, handles: Handles, hparams: Hparams) -> None: 
398    train_loss = sess.graph.get_tensor_by_name(f'{handles.train_loss}:0') 
399    optimize = sess.graph.get_operation_by_name(f'{handles.optimize}') 
400   
401    for i_epoch in range(1, hparams.num_epochs + 1): 
402      time_begin = time.monotonic() 
403      train_loss_vals = [] 
404      for feed in get_feed(all_batches=train_batches, graph=sess.graph, handles=handles): 
405        train_loss_val, _ = sess.run([train_loss, optimize], feed) 
406        train_loss_vals.append(train_loss_val) 
407   
408      val_loss_vals = [] 
409      for feed in get_feed(all_batches=val_batches, graph=sess.graph, handles=handles): 
410        val_loss_val = sess.run(train_loss, feed) 
411        val_loss_vals.append(val_loss_val) 
412   
413      train_loss_val = np.mean(train_loss_vals[-len(val_loss_vals):]) 
414      val_loss_val = np.mean(val_loss_vals) 
415   
416      time_end = time.monotonic() 
417      logger.info(' '.join([ 
418        f'epoch={i_epoch:0{len(str(hparams.num_epochs))}d}/{hparams.num_epochs}', 
419        f'train_loss={train_loss_val:.4f}', 
420        f'val_loss={val_loss_val:.4f}', 
421        f'duratoin={time_end-time_begin:.4f}s'])) 
422   
423    save_model(sess, hparams.checkpoint_prefix)

467  def translate(input_ids_var_length, handles, hparams): 
468    graph = tf.Graph() 
469    with tf.Session(graph=graph) as sess: 
470      restore_model(sess, hparams.checkpoint_prefix) 
471   
472      input_ids, input_lengths = pad_right(input_ids_var_length, vocab_to_id) 
473      feed = {graph.get_tensor_by_name(f'{handles.batch_size}:0'): len(input_ids_var_length), 
474              graph.get_tensor_by_name(f'{handles.input_ids}:0'): input_ids, 
475              graph.get_tensor_by_name(f'{handles.input_lengths}:0'): input_lengths, 
476              graph.get_tensor_by_name(f'{handles.max_decode_iterations}:0'): 2 * np.max(input_lengths)} 
477   
478      infer_logits = graph.get_tensor_by_name('infer_logits:0') 
479      logits_val = sess.run(infer_logits, feed) 
480      target_ids = np.argmax(logits_val, axis=2) 
481   
482      return target_ids

486    return [[d[k] for k in seq] for seq in sequences]

493  # All seen training and validating examples. 
494  def all_seen(batches, vocab_to_id, id_to_vocab): 
495    seen = [] 
496    for batch in batches: 
497      seqs = batch[0] 
498      for seq in seqs: 
499        filling_left = True 
500        left, right = [], [] 
501        for id_ in seq: 
502          if filling_left: 
503            if id_ == vocab_to_id['+']: 
504              filling_left = False 
505            else: 
506              left.append(id_) 
507          else: 
508            if id_ == vocab_to_id['<PAD>']: 
509              break 
510            else: 
511              right.append(id_) 
512        seen.append((ids_to_int(left, id_to_vocab), ids_to_int(right, id_to_vocab))) 
513    return seen

516  def make_unseen(hparams, seen, n): 
517    made = 0 
518    res = [] 
519    while made < n: 
520      while True: 
521        x, y = np.random.randint(hparams.data_lower, hparams.data_upper, (2,)) 
522        if (x, y) not in seen: 
523          res.append((x, y)) 
524          made += 1 
525          break 
526    return res
